def cannot_be(op, obj)
  assert("#{op} is not a comparison operator.") do
    [:==, :!=, :<, :>, :<=, :>=].any? { |o| o == op }
  end
  must_respond_to(op)
  refute("#{self.inspect} is #{op} #{obj.inspect}.") do
    self.send(op, obj)
  end
end
