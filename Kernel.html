def cannot_raise(exception = StandardError)
  yield
rescue exception
  false
rescue Exception => ex
  raise ex
else
  true
end

def capture_output
  stream = ''
  def stream.write(data)
    self << data
  end
  $stdout = stream
  yield
  $stdout = STDOUT
  stream
end
